# Production Dependencies for Car Damage Detection System
# 
# This file lists all Python packages required to run the car damage detection
# system. Install with: pip install -r requirements.txt
#

# - Provides training, inference, and utility functions
# - Official implementation from Ultralytics

# Streamlit - Web application framework
# - Create interactive web UI without HTML/CSS/JavaScript
# - File upload widget, image display, real-time updates
# - Auto-server, no Flask/Django boilerplate needed
streamlit>=1.28.0

# PyTorch - Deep learning framework
# - Tensor computation and autograd (automatic differentiation)
# - GPU acceleration via CUDA (if available)
# - Foundation for YOLOv8 model implementation

# - Image transformation and preprocessing

# - Read/write images and videos
# - Image resizing, color space conversion

# NumPy - Numerical computing
